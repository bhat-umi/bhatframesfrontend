  let itemsArray = [];

  // Add Item to the Table
  function addItem() {
      const description = document.getElementById('itemDescription').value;
      const quantity = parseInt(document.getElementById('itemQuantity').value);
      const rate = parseFloat(document.getElementById('itemRate').value);
  
      if (description && quantity > 0 && rate > 0) {
          const amount = quantity * rate;
          itemsArray.push({ description, quantity, rate, amount });
          updateItemsTable();
          clearItemInputs();
      } else {
          alert("Please enter valid item details.");
      }
  }
  
  // Update the Items Table inside Modal
  function updateItemsTable() {
      const items = document.getElementById('items').getElementsByTagName('tbody')[0];
      items.innerHTML = ''; // Clear current table rows
      itemsArray.reverse(); 
      itemsArray.forEach((item, index) => {
          const row = `<tr>
                          <td>${item.description}</td>
                          <td>${item.quantity}</td>
                          <td>₹${item.rate.toFixed(2)}</td>
                         
                          <td><button class="btn btn-danger btn-remove" onclick="removeItem(${index})"><i class="bi bi-trash"></i> </button></td>
                      </tr>`;
          items.innerHTML += row;
      });
  }
  
  // Remove Item from the Table
  function removeItem(index) {
      itemsArray.splice(index, 1);
      updateItemsTable();
  }
  
  // Clear Input Fields After Adding Item
  function clearItemInputs() {
      document.getElementById('itemDescription').value = '';
      document.getElementById('itemQuantity').value = '';
      document.getElementById('itemRate').value = '';
  }

        // JavaScript for searching customers as user types
        const customerInput = document.getElementById('payment-customer');
        const customerList = document.getElementById('customer-list');
        
        // Example customers, this should ideally come from a server-side API
        const customers = [
            { name: 'Asif', phone: '123-456-7890' },
            { name: 'Three way', phone: '987-654-3210' },
            { name: 'Moile Hut', phone: '555-555-5555' }
        ];
    
        customerInput.addEventListener('input', function() {
            const searchTerm = customerInput.value.toLowerCase();
            const filteredCustomers = customers.filter(customer =>
                customer.name.toLowerCase().includes(searchTerm) ||
                customer.phone.includes(searchTerm)
            );
    
            customerList.innerHTML = ''; // Clear previous results
    
            if (filteredCustomers.length > 0) {
                customerList.style.display = 'block';
                filteredCustomers.forEach(customer => {
                    const li = document.createElement('li');
                    li.classList.add('list-group-item');
                    li.textContent = `${customer.name} (${customer.phone})`;
                    li.onclick = function() {
                        customerInput.value = customer.name; // Set selected customer name
                        customerList.style.display = 'none'; // Hide the list
                    };
                    customerList.appendChild(li);
                });
            } else {
                customerList.style.display = 'none'; // Hide if no match
            }
        });
    
        // Close the dropdown if clicked outside
        document.addEventListener('click', function(event) {
            if (!customerInput.contains(event.target) && !customerList.contains(event.target)) {
                customerList.style.display = 'none';
            }
        });
    </script>
